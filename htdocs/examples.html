   editor = {Robinson, W. F. and Huxtable, C. R. R.},

   address = {Cambridge},
}

Robinson, W. F. & Huxtable, C. R. R. (eds) 1988,
 \textit{Clinicopathologic Principles For Veterinary Medicine},
 Cambridge University Press, Cambridge.

\bibitem[huf]
Huffman, L. M. 1996, `Processing whey protein for use as a food
 ingredient' in \textit{Food Technology}, vol.50, no.2, pp.49-52.

\end{biblist}
